\begin{frame}[fragile]
\frametitle{Beamer slides can have code chucks}
Slides with code chunks need to be marked as "fragile"
\begin{Schunk}
\begin{Sinput}
> set.seed(1)
> x <- rnorm(100)
> mean(x)
\end{Sinput}
\begin{Soutput}
[1] 0.1088874
\end{Soutput}
\end{Schunk}
\end{frame}
